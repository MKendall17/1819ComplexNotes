\begin{definition}
$\C := \mathbb{R}^2 \setminus (\R \times \{0\}) \cup \R$.

\begin{definition}
    $i := (0,1)$.

\begin{theorem}
    The only linear transformations that
    \begin{enumerate}
        \item fix $(0,0)$
        \item preserve distance
        \item preserve orientation
    \end{enumerate}
    are of the form $(x,y) \mapsto (ax - by, bx + ay)$ where $a^2 + b^2 = 1$.
\end{theorem}
